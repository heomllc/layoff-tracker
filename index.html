        // Function to render layoff data into the DOM

        function renderLayoffs(layoffsToRender) {

            const layoffUpdatesDiv = document.getElementById('layoff-updates');

            layoffUpdatesDiv.innerHTML = ''; // Clear existing content


            if (layoffsToRender.length === 0) {

                layoffUpdatesDiv.innerHTML = '<p class="text-center text-gray-500 py-8">No layoff announcements found matching your criteria.</p>';

                return;

            }


            layoffsToRender.forEach(layoff => {

                const div = document.createElement('div');

                // Determine background color based on industry for visual variety

                let bgColorClass = 'bg-blue-50 border-blue-200 text-blue-800';

                if (layoff.industry.toLowerCase().includes('tech')) {

                    bgColorClass = 'bg-indigo-50 border-indigo-200 text-indigo-800';

                } else if (layoff.industry.toLowerCase().includes('retail')) {

                    bgColorClass = 'bg-red-50 border-red-200 text-red-800';

                } else if (layoff.industry.toLowerCase().includes('financial')) {

                    bgColorClass = 'bg-green-50 border-green-200 text-green-800';

                }


                div.className = `p-4 rounded-lg border ${bgColorClass}`;

                div.innerHTML = `

                    <p class="text-lg font-medium">

                        <span class="font-bold">${layoff.company}</span> announced <span class="text-red-600">${layoff.employees} layoffs</span> on ${layoff.date}.

                    </p>

                    <p class="text-sm text-gray-600 mt-1">

                        Reason: ${layoff.reason}. (Source: ${layoff.source})

                    </p>

                    <p class="text-xs text-gray-500 mt-1">

                        Industry: ${layoff.industry} | Location: ${layoff.location}

                    </p>

                `;

                layoffUpdatesDiv.appendChild(div);

            });

        }


        // Function to apply filters

        function applyFilters() {

            const companyName = document.getElementById('company-name-filter').value.toLowerCase();

            const industry = document.getElementById('industry-filter').value.toLowerCase();

            const location = document.getElementById('location-filter').value.toLowerCase();

            const startDate = document.getElementById('start-date-filter').value;

            const endDate = document.getElementById('end-date-filter').value;


            let filteredLayoffs = layoffData.filter(layoff => {

                const layoffDate = new Date(layoff.date); // Convert layoff date string to Date object


                // Company Name Filter

                const matchesCompany = companyName === '' || layoff.company.toLowerCase().includes(companyName);


                // Industry Filter

                const matchesIndustry = industry === '' || layoff.industry.toLowerCase().includes(industry);


                // Location Filter

                const matchesLocation = location === '' || layoff.location.toLowerCase().includes(location);


                // Date Range Filter

                const matchesStartDate = startDate === '' || layoffDate >= new Date(startDate);

                const matchesEndDate = endDate === '' || layoffDate <= new Date(endDate);


                return matchesCompany && matchesIndustry && matchesLocation && matchesStartDate && matchesEndDate;

            });


            // Sort filtered layoffs by date, newest first

            filteredLayoffs.sort((a, b) => new Date(b.date) - new Date(a.date));


            renderLayoffs(filteredLayoffs);

        }


        // Event listener for the Apply Filters button

        document.getElementById('apply-filters-btn').addEventListener('click', applyFilters);


        // Initial render when the page loads

        document.addEventListener('DOMContentLoaded', () => {

            // Sort initial data by date, newest first

            const sortedInitialData = [...layoffData].sort((a, b) => new Date(b.date) - new Date(a.date));

            renderLayoffs(sortedInitialData);

        });
